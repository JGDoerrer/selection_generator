% Define the sub command
\newcommand{\sub}[2]{
  \setcounter{minuteTimetag}{\value{minuteTimetag} - #1}
  \setcounter{secondTimetag}{\value{secondTimetag} - #2}
  \ifnum\value{secondTimetag}<0
    \setcounter{minuteTimetag}{\value{minuteTimetag} - 1}
    \setcounter{secondTimetag}{\value{secondTimetag} + 60}
  \fi
}

\newcommand{\printTimetag}{%
\the\value{minuteTimetag}:\ifnum\value{secondTimetag}<10{0}\fi\thesecondTimetag
}

\newcommand{\framewithnote}[5]{%
  \sub{#1}{#2}
  \begin{frame}{#3}
    #4
    \note{
      \fontsize{30}{30}\selectfont
      \begin{itemize}\color{white}
        {
          \fontsize{70}{80}\selectfont
          \item \parbox{\textwidth}{
            \centering
            \textbf{\printTimetag}
          }
        }
        #5
      \end{itemize}
    }
  \end{frame}
}
